function doGet(e) { return handleRequest(e); }
function doPost(e) { return handleRequest(e); }

function json(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

function handleRequest(e) {
  try {
    const p = e.parameter;
    const action = p.action;
    const ss = SpreadsheetApp.getActiveSpreadsheet();

    if (action === 'initHeaders') {
      const name = p.sheet;
      const headers = JSON.parse(p.headers);
      let sh = ss.getSheetByName(name);
      if (!sh) sh = ss.insertSheet(name);
      if (sh.getLastRow() === 0) sh.appendRow(headers);
      return json({ ok: true });
    }

    if (action === 'getAll') {
      const sh = ss.getSheetByName(p.sheet);
      if (!sh || sh.getLastRow() < 2) return json({ ok: true, data: [] });
      const rows = sh.getDataRange().getValues();
      const headers = rows[0];
      const data = rows.slice(1).map(r => {
        const obj = {};
        headers.forEach((h, i) => obj[h] = r[i]);
        return obj;
      });
      return json({ ok: true, data });
    }

    if (action === 'append') {
      const sh = ss.getSheetByName(p.sheet);
      if (!sh) return json({ ok: false, error: 'Sheet not found' });
      const row = JSON.parse(p.row);
      const headers = sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0];
      sh.appendRow(headers.map(h => row[h] !== undefined ? row[h] : ''));
      return json({ ok: true });
    }

    if (action === 'update') {
      const sh = ss.getSheetByName(p.sheet);
      if (!sh) return json({ ok: false, error: 'Sheet not found' });
      const row = JSON.parse(p.row);
      const idCol = p.idCol || 'id';
      const data = sh.getDataRange().getValues();
      const headers = data[0];
      const idIdx = headers.indexOf(idCol);
      for (let i = 1; i < data.length; i++) {
        if (String(data[i][idIdx]) === String(row[idCol])) {
          headers.forEach((h, j) => {
            if (row[h] !== undefined) sh.getRange(i+1, j+1).setValue(row[h]);
          });
          return json({ ok: true });
        }
      }
      return json({ ok: false, error: 'Row not found' });
    }

    if (action === 'delete') {
      const sh = ss.getSheetByName(p.sheet);
      if (!sh) return json({ ok: false, error: 'Sheet not found' });
      const idCol = p.idCol || 'id';
      const idVal = p.idVal;
      const data = sh.getDataRange().getValues();
      const headers = data[0];
      const idIdx = headers.indexOf(idCol);
      for (let i = 1; i < data.length; i++) {
        if (String(data[i][idIdx]) === String(idVal)) {
          sh.deleteRow(i + 1);
          return json({ ok: true });
        }
      }
      return json({ ok: false, error: 'Row not found' });
    }

    return json({ ok: false, error: 'Unknown action: ' + action });
  } catch (err) {
    return json({ ok: false, error: err.toString() });
  }
}
